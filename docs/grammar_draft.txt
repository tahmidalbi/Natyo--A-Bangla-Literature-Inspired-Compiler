Natyo Grammar Draft

program
    -> prostabona global_items uposhonghar

global_items
    -> global_item
    -> global_items global_item

global_item
    -> statement
    -> function_definition

statement
    -> declaration ;
    -> assignment ;
    -> print_statement ;
    -> input_statement ;
    -> return_statement ;
    -> function_call ;
    -> if_statement
    -> loop_statement

declaration
    -> shonkhya IDENTIFIER
    -> shonkhya IDENTIFIER = expression
    -> purnoshonkhya IDENTIFIER
    -> purnoshonkhya IDENTIFIER = expression
    -> bakkyo IDENTIFIER
    -> bakkyo IDENTIFIER = STRING_LITERAL
    -> okkhyor IDENTIFIER
    -> okkhyor IDENTIFIER = CHAR_LITERAL

assignment
    -> IDENTIFIER = expression
    -> IDENTIFIER = STRING_LITERAL
    -> IDENTIFIER = CHAR_LITERAL

print_statement
    -> shobdo expression
    -> shobdo STRING_LITERAL
    -> shobdo CHAR_LITERAL

input_statement
    -> jobdo IDENTIFIER

return_statement
    -> pherot expression
    -> pherot STRING_LITERAL
    -> pherot CHAR_LITERAL

if_statement
    -> jodi ( condition ) statement
    -> jodi ( condition ) statement onnotha statement

loop_statement
    -> chokro ( assignment ; condition ; assignment ) statement

function_definition
    -> drissho IDENTIFIER ( parameter_list_opt ) statement

function_call
    -> IDENTIFIER ( argument_list_opt )

expression
    -> expression + expression
    -> expression - expression
    -> expression * expression
    -> expression / expression
    -> expression % expression
    -> NUMBER
    -> IDENTIFIER
    -> ( expression )
    -> shakti ( expression , expression )
    -> borgomul ( expression )
